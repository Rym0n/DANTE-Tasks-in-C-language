void show(const void* ptr, size_t count)
{
    if (!ptr) return;
    for (size_t i = 0; i < count; ++i) {
        unsigned char c = *((char*)ptr + i);
        printf("%02x ", c);
    }
}
void inverse_bits(void* ptr, size_t offset, size_t count)
{
    if (!ptr) return;
    for (size_t i = 0; i < count; ++i) {
        *((char*)ptr + offset + i) = ~*((char*)ptr + offset + i);
    }
}
